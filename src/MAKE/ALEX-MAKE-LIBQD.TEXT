$EXEC{Builds the LIBQD library}{Before we build QuickDraw itself, make the FixMove program}{that fixes the move instructions in lines.obj}F{ilemgr}D{elete}LIBQD/FIXMOVE.OBJY{es}Q{uit}$SUBMIT BUILD/COMP(LIBQD/FIXMOVE)L{inker}LIBQD/FIXMOVE {Object code}IOSPASLIB               {Libraries}                      {No more input files}                      {No listing file}LIBQD/FIXMOVE.OBJ     {Output executable}$SUBMIT ALEX/ASM/LIBQD$SUBMIT ALEX/COMP/LIBQD{Now build the QuickDraw sample program}F{ilemgr}D{elete}LIBQD/QDSAMPLE.OBJY{es}Q{uit}$SUBMIT BUILD/COMP(LIBQD/QDSAMPLE)L{inker}LIBQD/QDSAMPLE {Object code}IOSPASLIB               {Libraries}IOSFPLIBSYS1LIB                       {No more input files}                       {No listing file}LIBQD/QDSAMPLE.OBJ     {Output executable}$ENDEXEC
